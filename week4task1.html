Reverse Number JavaScript Function

Here is the JavaScript function that reverses a number:

<script>
    function reverseNumber(num) {
        let numStr = num.toString();
        let reversedStr = numStr.split('').reverse().join('');
        let reversedNum = parseInt(reversedStr, 10);
        return reversedNum;
    }

    let x = 32243;
    let reversedX = reverseNumber(x);
    console.log(reversedX); // Expected Output: 34223
</script>
